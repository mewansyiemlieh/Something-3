-- HTTP Request Handler (supports multiple executors)
local function HttpRequest(url, method, headers, body)
    local success, response
    
    -- Try different HTTP methods based on executor
    if syn and syn.request then
        success, response = pcall(function()
            return syn.request({
                Url = url,
                Method = method,
                Headers = headers,
                Body = body
            })
        end)
    elseif http and http.request then
        success, response = pcall(function()
            return http.request({
                Url = url,
                Method = method,
                Headers = headers,
                Body = body
            })
        end)
    elseif request then
        success, response = pcall(function()
            return request({
                Url = url,
                Method = method,
                Headers = headers,
                Body = body
            })
        end)
    elseif http_request then
        success, response = pcall(function()
            return http_request({
                Url = url,
                Method = method,
                Headers = headers,
                Body = body
            })
        end)
    elseif HttpService and HttpService.RequestAsync then
        success, response = pcall(function()
            return HttpService:RequestAsync({
                Url = url,
                Method = method,
                Headers = headers,
                Body = body
            })
        end)
    else
        return nil, "No HTTP method available"
    end
    
    if success and response then
        return response
    else
        return nil, response
    end
end

-- ============ DRAGGING FUNCTIONALITY ============
local function MakeDraggable(frame, handle)
    local dragging = false
    local dragInput, dragStart, startPos
    
    handle = handle or frame
    
    handle.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = input.Position
            startPos = frame.Position
            
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)
    
    handle.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
            dragInput = input
        end
    end)
    
    UserInputService.InputChanged:Connect(function(input)
        if input == dragInput and dragging then
            local delta = input.Position - dragStart
            frame.Position = UDim2.new(
                startPos.X.Scale, startPos.X.Offset + delta.X,
                startPos.Y.Scale, startPos.Y.Offset + delta.Y
            )
        end
    end)
end

-- ============ AI RESPONSE GENERATION ============
function GenerateResponse(userMessage)
    if Config.APIKey == "" then
        return nil, "No API key configured"
    end
    
    -- Build messages array with context
    local messages = {}
    
    -- System prompt with personality
    if Config.PersonalityEnabled and Config.Personality ~= "" then
        table.insert(messages, {
            role = "system",
            content = Config.Personality .. "\n\nYou are responding in a Roblox game chat. Keep responses short and chat-appropriate. Your name is " .. LocalPlayer.Name .. "."
        })
    else
        table.insert(messages, {
            role = "system",
            content = "You are responding in a Roblox game chat as " .. LocalPlayer.Name .. ". Keep responses very short (1-2 sentences max)."
        })
    end
    
    -- Add chat history for context
    local historyStart = math.max(1, #ChatHistory - Config.ContextMemory + 1)
    for i = historyStart, #ChatHistory do
        table.insert(messages, ChatHistory[i])
    end
    
    -- Add current message
    table.insert(messages, {role = "user", content = userMessage})
    
    -- Determine endpoint and headers based on provider
    local endpoint, headers
    
    if Config.APIProvider == "OpenAI" then
        endpoint = "[api.openai.com](https://api.openai.com/v1/chat/completions)"
        headers = {
            ["Content-Type"] = "application/json",
            ["Authorization"] = "Bearer " .. Config.APIKey
        }
    elseif Config.APIProvider == "OpenRouter" then
        endpoint = "[openrouter.ai](https://openrouter.ai/api/v1/chat/completions)"
        headers = {
            ["Content-Type"] = "application/json",
            ["Authorization"] = "Bearer " .. Config.APIKey,
            ["HTTP-Referer"] = "[roblox.com](https://roblox.com)",
            ["X-Title"] = "Roblox AI Chatbot"
        }
    elseif Config.APIProvider == "Groq" then
        endpoint = "[api.groq.com](https://api.groq.com/openai/v1/chat/completions)"
        headers = {
            ["Content-Type"] = "application/json",
            ["Authorization"] = "Bearer " .. Config.APIKey
        }
    elseif Config.APIProvider == "Custom" then
        endpoint = Config.CustomEndpoint
        headers = {
            ["Content-Type"] = "application/json",
            ["Authorization"] = "Bearer " .. Config.APIKey
        }
    end
    
    local body = HttpService:JSONEncode({
        model = Config.APIProvider == "OpenRouter" and "openai/gpt-3.5-turbo" or 
                Config.APIProvider == "Groq" and "llama3-8b-8192" or "gpt-3.5-turbo",
        messages = messages,
        max_tokens = Config.MaxTokens,
        temperature = Config.Temperature
    })
    
    local response, err = HttpRequest(endpoint, "POST", headers, body)
    
    if response and response.Body then
        local success, data = pcall(function()
            return HttpService:JSONDecode(response.Body)
        end)
        
        if success and data.choices and data.choices[1] then
            local reply = data.choices[1].message.content
            
            -- Store in history
            table.insert(ChatHistory, {role = "user", content = userMessage})
            table.insert(ChatHistory, {role = "assistant", content = reply})
            
            -- Trim history if too long
            while #ChatHistory > Config.ContextMemory * 2 do
                table.remove(ChatHistory, 1)
            end
            
            return reply
        else
            return nil, "Failed to parse response: " .. tostring(data and data.error and data.error.message or "Unknown error")
        end
    else
        return nil, "HTTP request failed: " .. tostring(err)
    end
end

-- ============ CHAT HANDLING ============
local function ShouldRespond(playerName, message)
    if playerName == LocalPlayer.Name then return false end
    if not Config.Enabled then return false end
    
    local lowerMsg = string.lower(message)
    local myName = string.lower(LocalPlayer.Name)
    
    if Config.TriggerMode == "All" then
        return true
    elseif Config.TriggerMode == "Mention" then
        -- Check if mentioned by name or display name
        return string.find(lowerMsg, myName) or 
               string.find(lowerMsg, string.lower(LocalPlayer.DisplayName))
    elseif Config.TriggerMode == "Keyword" then
        for _, keyword in ipairs(Config.TriggerKeywords) do
            if string.find(lowerMsg, keyword) then
                return true
            end
        end
        return false
    end
    
    return false
end

local function SendChatMessage(message)
    -- Try different chat methods
    local success = false
    
    -- Method 1: TextChatService (modern)
    local TextChatService = game:GetService("TextChatService")
    local channels = TextChatService:FindFirstChild("TextChannels")
    if channels then
        local rbxGeneral = channels:FindFirstChild("RBXGeneral")
        if rbxGeneral then
            pcall(function()
                rbxGeneral:SendAsync(message)
                success = true
            end)
        end
    end
    
    -- Method 2: Legacy chat
    if not success then
        local ReplicatedStorage = game:GetService("ReplicatedStorage")
        local chatRemote = ReplicatedStorage:FindFirstChild("DefaultChatSystemChatEvents")
        if chatRemote then
            local sayEvent = chatRemote:FindFirstChild("SayMessageRequest")
            if sayEvent then
                pcall(function()
                    sayEvent:FireServer(message, "All")
                    success = true
                end)
            end
        end
    end
    
    -- Method 3: Direct chat
    if not success then
        pcall(function()
            game:GetService("Chat"):Chat(LocalPlayer.Character, message)
        end)
    end
end

local function ProcessChat(playerName, message)
    if ProcessingMessage then return end
    if not ShouldRespond(playerName, message) then return end
    
    ProcessingMessage = true
    StatusLabel.Text = "Status: Processing...\nFrom: " .. playerName
    
    task.spawn(function()
        -- Random delay for natural feel
        local delay = Config.ResponseDelay.Min + math.random() * (Config.ResponseDelay.Max - Config.ResponseDelay.Min)
        task.wait(delay)
        
        local contextMessage = playerName .. " said: " .. message
        local response, err = GenerateResponse(contextMessage)
        
        if response then
            -- Clean up response for chat
            response = response:gsub("\n", " "):gsub("%s+", " "):sub(1, 200)
            SendChatMessage(response)
            MessagesProcessed = MessagesProcessed + 1
            StatusLabel.Text = "Status: ✅ Responded\nMessages: " .. MessagesProcessed
        else
            StatusLabel.Text = "Status: ❌ Error\n" .. tostring(err):sub(1, 40)
        end
        
        ProcessingMessage = false
    end)
end

        end
    end)
